        const gridArea = document.getElementById('gridArea');

        for(let i = 0; i < itemCount; i++) {

            let dot = document.createElement('div');

            dot.classList.add('item-dot');

            

            // 随机分布在网格中

            let randomX = Math.random() * 98; // 0% - 98%

            let randomY = Math.random() * 95; // 0% - 95%

            

            // 随机高度和颜色变化模拟不同材料

            let randomHeight = 10 + Math.random() * 20; 

            

            dot.style.left = `${randomX}%`;

            dot.style.top = `${randomY}%`;

            dot.style.height = `${randomHeight}px`;


            // 偶尔有些是灰色的，代表不同类型的归档

            if (Math.random() > 0.7) {

                dot.style.backgroundColor = '#999';

            }


            gridArea.appendChild(dot);

        }
